CREATE TABLE AE_DB_OS_STAT (
DATA_TYPE VARCHAR( 20 ),
SNAP_ID NUMERIC NULL,
INSTANCE_NUMBER NUMERIC NULL,
INSTANCE_NAME VARCHAR( 48 ) NULL,
BEGIN_TIME TIMESTAMP NULL,
END_TIME TIMESTAMP NULL,
OS_STAT_NAME VARCHAR( 64 ) NULL,
VALUE NUMERIC NULL,
identifier VARCHAR (50) NULL,
create_dt timestamp with time zone default current_timestamp,
create_id varchar(20) default 'system' not null
)
