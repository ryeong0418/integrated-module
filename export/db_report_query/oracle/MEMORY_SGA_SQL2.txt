WITH BASE AS (SELECT INST_ID, ESTD_DB_TIME, ESTD_PHYSICAL_READS
                FROM GV$SGA_TARGET_ADVICE
               WHERE SGA_SIZE_FACTOR = 1)
SELECT X.INST_ID,
	   ROUND(X.SGA_SIZE/1024,1) || 'GB' SGA_SIZE,
	   X.PHYSICAL_READS || '%' PHYSICAL_READS
  FROM (SELECT A.INST_ID,
			   A.SGA_SIZE,
			   ROUND((B.ESTD_PHYSICAL_READS - A.ESTD_PHYSICAL_READS)*100/B.ESTD_PHYSICAL_READS,2) AS PHYSICAL_READS,
			   ROW_NUMBER() OVER(PARTITION BY A.INST_ID ORDER BY B.ESTD_PHYSICAL_READS - A.ESTD_PHYSICAL_READS DESC, A.SGA_SIZE ASC) RNK
		  FROM GV$SGA_TARGET_ADVICE A, BASE B
		 WHERE A.INST_ID = B.INST_ID
		   AND A.SGA_SIZE_FACTOR >= 1) X
WHERE X.RNK = 1
