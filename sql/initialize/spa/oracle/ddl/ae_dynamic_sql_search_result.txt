create table ae_dynamic_sql_search_result(
	source_sql_uid varchar(40) not null,
	analyzed_sel_cols text null,
	analyzed_table_names text null,
    analyzed_join_cols text null,
	analyzed_where_cols text null,
	target_sql_uid varchar(40) null,
	target_sql_id varchar(50) null,
	target_where_cols text null,
	dynamic_pattern text null,
	avg_lio float4 null,
    avg_pio float4 null,
	avg_elapsed_time float4 null,
	exec int8 null,
	create_dt timestamp with time zone default current_timestamp,

    constraint pk_ae_dynamic_sql_search_result primary key (source_sql_uid, target_sql_uid)
);
