        time,
        start_time,
        start_time + txn_elapse * '1 millisecond'::interval as end_time,
        txn_id,
        was_id,
        txn_elapse,
        txn_cpu_time,
        thread_memory,
        sql_exec_count,
        sql_elapse,

        fetch_time,

        exception,
        remote_count,
